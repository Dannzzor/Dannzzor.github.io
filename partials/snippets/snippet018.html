function SelectSingleNode(xmlDoc, elementPath) {
    if (xmlDoc.evaluate) {
        var nodes = xmlDoc.evaluate(elementPath, xmlDoc, null, XPathResult.ANY_TYPE, null);
        var results = nodes.iterateNext();
        return results;
    } else {
        return xmlDoc.selectSingleNode(elementPath);
	}
}

Usage:

// replace this
var selectedNode = xmlDoc.selectSingleNode(elementPath);

// with this
var selectedNode = SelectSingleNode(xmlDoc, elementPath);

Comments:

This was recently used to update DbNetGrid to work with the latest Google Chrome patch on 04/09/2014 which removed the depreciated document.selectSingleNode() method.
