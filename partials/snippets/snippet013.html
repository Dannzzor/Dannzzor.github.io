function compareDates( date1, comparisonType, date2 ) {
    var result = false;
    
    // all three parameters requierd
    if( !date1 || !date2 || !comparisonType ) {
        return false;
    }

    // convert dates to date objects, then strip the time from the dates for precise comparison
    date1 = new Date(date1).setHours(0,0,0,0);
    date2 = new Date(date2).setHours(0,0,0,0);

    // make sure dates were passed
    if( isNaN(date1) || isNaN(date2) ){
        return false;
    }

    switch (comparisonType){
        case ">=" :
            result = date1 >= date2;
        break;
        case "<=" :
            result = date1 <= date2;
        break;
        case ">" :
            result = date1 > date2;
        break;
        case "<" :
            result = date1 < date2;
        break;
        case "==" :
            result = date1 == date2;
        break;

        default :
            return false;
    }
    return result;
}

Usage:

compareDates('4/05/2010', '>', '5/05/2010');            // false
compareDates('4/05/2010', '<', '5/05/2010');            // true
compareDates('4/05/2010', '==', '5/05/2010');           // false
compareDates('4/05/2010', '<=', '5/05/2010');           // true

var today = new Date();

compareDates( today , '>=', '5/05/2010');               // true
