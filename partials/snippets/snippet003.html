function getParams(url){
    var map = {};
    url = url || window.location.href;
    url.replace(/[?&]+([^=&]+)=([^&]*)/gi, function(m,key,value) {
        map[key] = value;
    });
    return map;
}

Usage:

var url = "http://www.whatever.com?something=23&somethingelse=bla";
var x = getParams();
x.something;			// 23
x.somethingelse;		// "bla"
x.doesntexist;			// undefined

Comments:

Used to grab all of the params from a URL, then return them as a key value map.
