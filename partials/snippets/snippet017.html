var locationArray = [
	{region:'c', country:'a'},
	{region:'d', country:'d'},
	{region:'b', country:'h'},
	{region:'a', country:'f'},
	{region:'a', country:'d'},
	{region:'a', country:'a'}
];

function doubleSort (a, b) {
    var aReg = a.region || '';
    var bReg = b.region || '';
    var aCnt = a.country || '';
    var bCnt = b.country || '';

    if(aReg.toLowerCase() == bReg.toLowerCase()) {
        return (aCnt.toLowerCase() < bCnt.toLowerCase()) ? -1 : (aCnt.toLowerCase() > bCnt.toLowerCase()) ? 1 : 0;
    } else {
        return (aReg.toLowerCase() < bReg.toLowerCase()) ? -1 : 1;
    }
}

var sortedArr = locationArray.sort(doubleSort);

JSON.stringify(sortedArr);           //[{region:'a', country:'a'},{region:'a', country:'d'},{region:'a', country:'f'},{region:'b', country:'h'},{region:'c', country:'a'},{region:'d', country:'d'}];
